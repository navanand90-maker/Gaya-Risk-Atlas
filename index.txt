        // --- 1. SUPER AUTO-DETECTOR ---
        function detectColumn(props, type) {
            // Priority list (case insensitive search)
            var candidates = ['Risk Class', 'Risk_Class', 'gridcode', 'Total', '_mean', 'Mean', 'MEAN', 'Pop', 'P_Total', '_sum', 'Sum', 'SUM'];
            var keys = Object.keys(props);
            
            // 1. Check exact/near matches
            for (var c of candidates) {
                var match = keys.find(k => k.toLowerCase() === c.toLowerCase());
                if (match) return match;
            }

            // 2. Scan for ANY number (excluding IDs)
            if (type === 'num' || type === 'lightning') {
                for (var key in props) {
                    var val = props[key];
                    if (typeof val === 'number' && !key.toLowerCase().includes('id') && !key.toLowerCase().includes('area')) {
                        return key;
                    }
                }
            }
            // 3. Last ditch effort for Text categories
            return 'Risk Class';
        }

        // --- 2. STATS ---
        function calculateStats(data, colName) {
            var min = Infinity, max = -Infinity;
            data.features.forEach(f => {
                var val = f.properties[colName];
                if (val !== undefined && val !== null && !isNaN(val)) {
                    val = parseFloat(val);
                    if (val < min) min = val;
                    if (val > max) max = val;
                }
            });
            if (min === Infinity) return { min: 0, max: 1 };
            if (min === max) return { min: min * 0.9, max: max * 1.1 };
            return { min: min, max: max };
        }

        // --- 3. UNIVERSAL COLORS ---
        
        // LIGHTNING (Hardcoded rules)
        function getLightningColor(d) {
            var num = parseFloat(d);
            if (isNaN(num)) return '#cccccc';
            if (num > 9) return '#4a148c'; 
            if (num >= 6) return '#d32f2f'; 
            if (num >= 3) return '#f57f17'; 
            return '#388e3c'; 
        }

        // FLOOD & DROUGHT (Text or Gridcode)
        function getCatColor(d) {
            if (!d) return '#ccc';
            // If it's a number (Gridcode 1-5)
            if (typeof d === 'number') {
                if (d >= 5) return '#4a148c'; // Very High
                if (d === 4) return '#d32f2f'; // High
                if (d === 3) return '#f57f17'; // Moderate
                return '#388e3c'; // Low/Very Low
            }
            // If it's text
            var val = d.toString().toLowerCase();
            if (val.includes('very high') || val.includes('severe') || val.includes('extreme')) return '#4a148c';
            if (val.includes('high')) return '#d32f2f';
            if (val.includes('mod') || val.includes('med')) return '#f57f17';
            if (val.includes('very low')) return '#2e7d32'; // Extra dark green for very low
            if (val.includes('low')) return '#388e3c';
            return '#ccc';
        }

        // DYNAMIC (Fire/Quake/Heat)
